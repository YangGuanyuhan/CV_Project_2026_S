\section{Method}
% ══════════════════════════════════════════════

\begin{frame}{3. Method — Model Architecture Overview}
  \begin{columns}[T]
    \column{0.48\textwidth}
    We reproduce the Grounding DINO inference pipeline using the official
    \texttt{groundingdino-py} package (Liu et al., ECCV 2024) without modifying the model
    architecture. Our work focuses on wrapping the official implementation with a
    modular Python interface for configurable inference, evaluation, and visualization.

    \vspace{0.15cm}
    \begin{block}{Scope \& Assumptions}
      \scriptsize
      \begin{itemize}
        \item Zero-shot only — no fine-tuning on COCO
        \item Fixed confidence thresholds
        \item Canonical COCO category names
        \item Single-scale inference (no test-time augmentation)
      \end{itemize}
    \end{block}

    \column{0.50\textwidth}
    \begin{block}{Three Core Components}
      \scriptsize
      \begin{enumerate}\setlength{\itemsep}{4pt}
        \item \textbf{Swin Transformer backbone} (Liu et al., ICCV 2021):\\
              Hierarchical visual features through 4 stages (stride $4/8/16/32$).\\
              Shifted-window attention — $O(N)$ complexity.\\
              Config: window=7, embed\_dim=96, depths=[2,2,6,2].
        \item \textbf{BERT language encoder} (Devlin et al., NAACL 2019):\\
              bert-base-uncased — 12 layers, hidden=768, vocab=30,522.\\
              Bidirectional self-attention for contextualized token embeddings.
        \item \textbf{Deformable attention fusion}:\\
              6 cross-modal layers, 8 heads each, 256-dim features.\\
              900 object queries $\times$ visual locations $\times$ text tokens.
      \end{enumerate}
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{3. Method — Inference Pipeline}
  \textbf{Stage-by-stage data flow from inputs to COCO-format detections:}

  \vspace{0.2cm}
  \begin{enumerate}\setlength{\itemsep}{6pt}
    \item \textbf{Model Loading}\\
          \texttt{GroundingDINOModel} class wraps \texttt{load\_model()} from
          \texttt{groundingdino-py}. Validates checkpoint (662 MB) and config file,
          detects CUDA/CPU device, exposes \texttt{@torch.no\_grad()} predict method.

    \item \textbf{Image Preprocessing}\\
          BGR $\rightarrow$ RGB (OpenCV $\rightarrow$ PIL convention).\\
          Resize: \texttt{RandomResize([800], max\_size=1333)} — shorter side to 800px,
          longer side constrained to $\le$ 1333px.\\
          Normalize: ImageNet stats (mean $=[0.485,0.456,0.406]$,
          std $=[0.229,0.224,0.225]$), HWC $\rightarrow$ CHW tensor.

    \item \textbf{Text Encoding \& Cross-Modal Fusion}\\
          BERT WordPiece tokenizer $\rightarrow$ $(1, L)$ input IDs + attention mask
          ($L \le 256$).\\
          Swin-T extracts multi-scale features $\oplus$ BERT produces $(1, L, 768)$
          embeddings.\\
          6 deformable fusion layers $\rightarrow$ each of 900 queries outputs
          $(cx,cy,w,h)_{\text{norm}}$, confidence $s$, phrase index.

    \item \textbf{Postprocessing}\\
          $(cx,cy,w,h)_{\text{norm}} \xrightarrow{\text{box\_cxcywh\_to\_xyxy}}
          (x_1,y_1,x_2,y_2)_{\text{pixel}} \xrightarrow{} (x,y,w,h)_{\text{COCO}}$.\\
          Filter: box\_threshold=0.35, text\_threshold=0.25 (default).\\
          Clamp boxes to image boundaries; discard degenerate boxes ($w \le 0$ or $h \le 0$).
  \end{enumerate}
\end{frame}

\begin{frame}{3. Method — Prompt Design \& Phrase-to-Category Mapping}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \begin{block}{Dot-Separated Prompt Format (Liu et al., ECCV 2024)}
      \small
      Enumerate all 80 COCO categories separated by \texttt{ . } (space-dot-space):
      \vspace{2pt}
      \begin{center}
        \texttt{"person . bicycle . car . motorcycle . airplane . bus . \ldots . toothbrush ."}
      \end{center}
      \vspace{2pt}
      \textbf{Why dot-separated?} BERT WordPiece tokenizer treats the period
      surrounded by spaces as an independent separator token, creating clear
      boundaries between adjacent category names. This prevents the tokenizer from
      merging category names into ambiguous subword units — a failure mode that
      occurs with comma-separated or sentence-style formats.
    \end{block}

    \column{0.43\textwidth}
    \begin{block}{Three-Layer Matching Strategy}
      \scriptsize
      \texttt{src/datasets/coco\_categories.py} maps each detected phrase to a
      COCO category ID (1--90):
      \begin{enumerate}\setlength{\itemsep}{4pt}
        \item \textbf{Exact match}: case-insensitive string equality
        \item \textbf{Substring match}: phrase $\subset$ category or vice versa\\
              (e.g., ``dining'' $\rightarrow$ ``dining table'')
        \item \textbf{Alias match}: manually curated dictionary of 21 aliases\\
              (e.g., ``tv''/``television'' $\rightarrow$ ``tv'', ``sofa'' $\rightarrow$ ``couch'')
      \end{enumerate}
      \vspace{3pt}
      Phrases failing all three layers ($\sim$2--5\%) are discarded.
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{4.1 Datasets — COCO 2017 Validation Set}
  \begin{columns}[T]
    \column{0.50\textwidth}
    \begin{block}{Dataset Overview (Lin et al., ECCV 2014)}
      \small
      \textbf{COCO 2017 validation set} (Common Objects in Context):
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item \textbf{Images}: 5,000 RGB images from diverse real-world scenes (indoor, outdoor, street views, crowded spaces)
        \item \textbf{Categories}: 80 common object classes (person, bicycle, car, dog, chair, \ldots)
        \item \textbf{Annotations}: 36,781 ground-truth bounding boxes in $[x, y, w, h]$ format, averaging $\sim$7.4 per image
        \item \textbf{Resolution}: Varying from $\sim$480$\times$640 to full HD
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{Key Characteristics}
      \small
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item \textbf{Long-tail distribution}: frequent classes (``person'', $\sim$262K training instances) coexist with rare classes (``hair drier'', $<$100 instances)
        \item \textbf{Category IDs}: Non-consecutive 1--90 with gaps
        \item \textbf{Annotation format}: COCO $[x, y, \text{width}, \text{height}]$ coordinate system
        \item \textbf{For ablation}: Fixed 500-image subset selected with random seed 42 for reproducibility
      \end{itemize}
    \end{block}

    \vspace{0.1cm}
    \begin{block}{Dataset Splits}
      \scriptsize
      \renewcommand{\arraystretch}{1.2}
      \begin{tabular}{lccc}
        \toprule
        \textbf{Split} & \textbf{Images} & \textbf{Annotations} & \textbf{Usage} \\
        \midrule
        train2017 & 118,287 & $\sim$860K & Model training / fine-tuning \\
        \textbf{val2017} & \textbf{5,000} & \textbf{36,781} & \textbf{Evaluation (this project)} \\
        test2017 & 40,670 & — & Official benchmark (no public GT) \\
        \bottomrule
      \end{tabular}
    \end{block}
  \end{columns}
\end{frame}

\section{Conclusion}
% ══════════════════════════════════════════════

\begin{frame}{5. Conclusion}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \begin{block}{Problem \& Approach}
      \small
      \textbf{Challenge}: Open-vocabulary object detection — detecting arbitrary
      text-described categories without task-specific training data.

      \vspace{3pt}
      \textbf{Method}: Reproduced Grounding DINO (Liu et al., ECCV 2024) using the official
      \texttt{groundingdino-py} inference backend with Swin-Tiny backbone in a
      zero-shot setting on COCO 2017 val2017.

      \vspace{3pt}
      \textbf{Key configuration}:
      \begin{itemize}\setlength{\itemsep}{1pt}\scriptsize
        \item Config-driven pipeline (OmegaConf YAML + CLI overrides)
        \item Dot-separated prompt format (all 80 COCO categories)
        \item box\_threshold=0.35, text\_threshold=0.25 (default)
        \item Single-scale inference, no fine-tuning
      \end{itemize}
    \end{block}

    \column{0.43\textwidth}
    \begin{block}{Core Findings}
      \small
      \begin{enumerate}\setlength{\itemsep}{4pt}
        \item Grounding DINO achieves viable zero-shot performance on COCO without task-specific training
        \item Threshold selection significantly impacts the precision-recall tradeoff
        \item Dot-separated prompt format is a \textbf{functional requirement} — alternative formats degrade severely
        \item Small-object detection remains the primary bottleneck (resolution constraint from Swin-T backbone)
        \item Localization quality lags behind recognition capability
      \end{enumerate}
    \end{block}
  \end{columns}

  \vspace{0.2cm}
  \begin{tikzpicture}
